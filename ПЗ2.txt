#include <iostream>
#include <cmath>

int main()
{
  int f, i, c = 0;
  string b;
  cin >> i;
  cout << i << "=";
  f = i;
  while (f){
    f = f/10;
    c++;
  }
  f = i;
  c--;
  while (abs(f) > 0) {
    if (f % 10 != 0){
      if (i>0) {
        cout <<"+"<< int(i/(pow(10, c)))%10<<"*10^"<<c--;

      }
    }
    f = f / 10;
  }
  return 0;

}
